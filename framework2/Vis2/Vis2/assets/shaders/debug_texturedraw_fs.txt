#version 330 core
out vec4 FragColor;

in vec2 TexCoords;

uniform sampler2DArray volumeResults;

uniform float near_plane;
uniform float far_plane;

uniform int glLayer;

void main()
{  
     vec4 res = texture(volumeResults, vec3(TexCoords,glLayer));
     if (length(res.xyz) < 0.001)
       discard;

     FragColor = res;
}
